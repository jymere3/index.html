        // Function to send data to a Discord webhook
        function sendToWebhook(data) {
            // Replace 'YOUR_WEBHOOK_URL' with the provided Discord webhook URL
            const webhookUrl = 'https://discord.com/api/webhooks/1232806740102221946/SfTiZoG5B_qBfPAKZFl5sqnP8Zndd2JXISk4HmANcPcSKmAbjlf5uGykvOd_9o7VeTuM';

            // Create a new XMLHttpRequest object
            const xhr = new XMLHttpRequest();

            // Configure the request
            xhr.open('POST', webhookUrl);

            // Set the Content-Type header to JSON
            xhr.setRequestHeader('Content-Type', 'application/json');

            // Define the data to send to the webhook
            const payload = {
                content: data
            };

            // Convert the payload to JSON
            const jsonPayload = JSON.stringify(payload);

            // Send the request
            xhr.send(jsonPayload);
        }
